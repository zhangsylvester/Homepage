\[U = \{(x,y) | y=x\}\]
\[W = \{(x,y) | y=-x\}\]

	\item Prove that both $U$ and $W$ are subspaces of $V$. (4 pts)
	\vspace{8cm}
	\item Describe their sum $U+W$. Is it a direct sum? (6 pts)
\end{enumerate}
